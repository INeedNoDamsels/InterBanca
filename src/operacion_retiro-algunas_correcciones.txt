def retiro():
    misc.interfaz.head()
    misc.interfaz.nombre_operacion(1)

    intentos = 0 # podríamos hacer una función que verifique la cantidad de intentos... :p

    clave = int(input(" >> Ingrese la clave de seguridad: "))
    try:
        assert clave == misc.globales.clave_a
        numero = int(input(" >> Ingrese número de cuenta: "))
        try:
            assert numero == misc.globales.clave_b

            tipo_cambio()

            monto = float(input("\n >> Ingrese monto: $"))
            try:
                assert 0 < monto <= misc.globales.saldo
                if misc.globales.moneda == "ARS":
                    misc.globales.dinero -= monto
                else:
                    misc.globales.dinero -= misc.globales.conversor_a_ars(monto)

                opcion = int(input("\n ¿Desea imprimir el voucher? (<1> Si <2> No): "))

                misc.interfaz.head()
                misc.interfaz.nombre_operacion(2)
                if opcion == 1:
                    print("\t\t\t\t Imprimiendo")
                    lapso(2)
                print("\t\t\t\t  Realizado")
            except:
                misc.interfaz.head()
                misc.interfaz.nombre_operacion(2)
                print("\t\t\t    Fondos insuficientes")
        except:
            misc.interfaz.head()
            misc.interfaz.nombre_operacion(2)
            print("\t\t\t     Cuenta inexistente")
    except:
        intentos += 1

        if intentos == 3:
            misc.globales.final("\n\t       Demasiados intentos fallidos, tarjeta retenida \
\n\t  Para más información, contacte a su proveedor de tarjetas")
        else:
            retiro()

    lapso(2)
    menu.opciones()
